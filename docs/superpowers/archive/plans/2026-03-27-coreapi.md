# CoreAPI Unified Extension Interface — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a unified CoreAPI Go interface that gives all extension types (Tengo, gRPC, future) equal bidirectional access to CMS capabilities, with manifest-declared capability-based security.

**Architecture:** Single `CoreAPI` interface in `internal/coreapi/` backed by GORM/EventBus/EmailDispatcher. Three adapters: Tengo (`core/*` modules replacing `cms/*`), gRPC (VibeCMSHost service via go-plugin GRPCBroker), and internal (direct calls). Capability guard wraps the implementation per-caller. Extension manifests declare required capabilities.

**Tech Stack:** Go 1.24, GORM, Fiber, Tengo v2, HashiCorp go-plugin (gRPC), Protocol Buffers

// internal/coreapi/api.go
package coreapi

import (
	"context"
	"io"
	"time"
)

// CoreAPI is the unified interface for all CMS operations.
// All extension types (Tengo, gRPC, internal) use this interface.
type CoreAPI interface {
	// Nodes
	GetNode(ctx context.Context, id uint) (*Node, error)
	QueryNodes(ctx context.Context, query NodeQuery) (*NodeList, error)
	CreateNode(ctx context.Context, input NodeInput) (*Node, error)
	UpdateNode(ctx context.Context, id uint, input NodeInput) (*Node, error)
	DeleteNode(ctx context.Context, id uint) error

	// Settings
	GetSetting(ctx context.Context, key string) (string, error)
	SetSetting(ctx context.Context, key, value string) error
	GetSettings(ctx context.Context, prefix string) (map[string]string, error)

	// Events
	Emit(ctx context.Context, action string, payload map[string]any) error
	Subscribe(ctx context.Context, action string, handler EventHandler) (UnsubscribeFunc, error)

	// Email
	SendEmail(ctx context.Context, req EmailRequest) error

	// Menus
	GetMenu(ctx context.Context, slug string) (*Menu, error)
	GetMenus(ctx context.Context) ([]*Menu, error)
	CreateMenu(ctx context.Context, input MenuInput) (*Menu, error)
	UpdateMenu(ctx context.Context, slug string, input MenuInput) (*Menu, error)
	DeleteMenu(ctx context.Context, slug string) error

	// Routes
	RegisterRoute(ctx context.Context, method, path string, meta RouteMeta) error
	RemoveRoute(ctx context.Context, method, path string) error

	// Filters
	RegisterFilter(ctx context.Context, name string, priority int, handler FilterHandler) (UnsubscribeFunc, error)
	ApplyFilters(ctx context.Context, name string, value any) (any, error)

	// Media
	UploadMedia(ctx context.Context, req MediaUploadRequest) (*MediaFile, error)
	GetMedia(ctx context.Context, id uint) (*MediaFile, error)
	QueryMedia(ctx context.Context, query MediaQuery) ([]*MediaFile, error)
	DeleteMedia(ctx context.Context, id uint) error

	// Users (read-only for extensions)
	GetUser(ctx context.Context, id uint) (*User, error)
	QueryUsers(ctx context.Context, query UserQuery) ([]*User, error)

	// HTTP (outbound)
	Fetch(ctx context.Context, req FetchRequest) (*FetchResponse, error)

	// Log
	Log(ctx context.Context, level, message string, fields map[string]any) error
}

package coreapi

import (
	"errors"

var (
	ErrCapabilityDenied = errors.New("capability denied")
	ErrNotFound         = errors.New("not found")
	ErrValidation       = errors.New("validation error")
	ErrInternal         = errors.New("internal error")
)

// APIError wraps a sentinel error with a message for richer context.
type APIError struct {
	Code    error
	Message string
}

func (e *APIError) Error() string {
	return fmt.Sprintf("%s: %s", e.Code, e.Message)
}

func (e *APIError) Unwrap() error {
	return e.Code
}

func NewCapabilityDenied(capability string) *APIError {
	return &APIError{Code: ErrCapabilityDenied, Message: fmt.Sprintf("missing capability: %s", capability)}
}

func NewNotFound(resource string, id any) *APIError {
	return &APIError{Code: ErrNotFound, Message: fmt.Sprintf("%s %v not found", resource, id)}
}

func NewValidation(msg string) *APIError {
	return &APIError{Code: ErrValidation, Message: msg}
}

	return &APIError{Code: ErrInternal, Message: msg}

// internal/coreapi/context.go
package coreapi

import "context"

type ctxKey int

const callerKey ctxKey = iota

// CallerInfo identifies who is calling the CoreAPI.
type CallerInfo struct {
	Slug         string          // extension slug, empty for internal
	Type         string          // "tengo", "grpc", "internal"
	Capabilities map[string]bool // allowed capabilities from manifest
}

// InternalCaller returns a CallerInfo for trusted internal code.
func InternalCaller() CallerInfo {
	return CallerInfo{Slug: "", Type: "internal", Capabilities: nil}
}

// WithCaller attaches CallerInfo to a context.
func WithCaller(ctx context.Context, caller CallerInfo) context.Context {
	return context.WithValue(ctx, callerKey, caller)
}

// CallerFromContext extracts CallerInfo from context.
// Returns InternalCaller if none set.
func CallerFromContext(ctx context.Context) CallerInfo {
	if c, ok := ctx.Value(callerKey).(CallerInfo); ok {
		return c
	}
	return InternalCaller()
}

type capabilityGuard struct {
	inner CoreAPI
}

// NewCapabilityGuard wraps a CoreAPI implementation with capability checks.
func NewCapabilityGuard(inner CoreAPI) CoreAPI {
	return &capabilityGuard{inner: inner}
}

func (g *capabilityGuard) check(ctx context.Context, capability string) error {
	caller := CallerFromContext(ctx)
	if caller.Type == "internal" {
		return nil
	}
	if caller.Capabilities[capability] {
		return nil
	}
	return NewCapabilityDenied(capability)
}

// internal/coreapi/impl_nodes.go
package coreapi

import (
	"context"
	"encoding/json"

	"vibecms/internal/models"
)

func (c *coreImpl) GetNode(ctx context.Context, id uint) (*Node, error) {
	node, err := c.contentSvc.GetByID(id)
	if err != nil {
		return nil, NewNotFound("node", id)
	}
	return nodeFromModel(node), nil
}

func (c *coreImpl) QueryNodes(ctx context.Context, query NodeQuery) (*NodeList, error) {
	tx := c.db.Model(&models.ContentNode{})

	if query.NodeType != "" {
		tx = tx.Where("node_type = ?", query.NodeType)
	}
	if query.Status != "" {
		tx = tx.Where("status = ?", query.Status)
	}
	if query.ParentID != nil {
		tx = tx.Where("parent_id = ?", *query.ParentID)
	}
	if query.LanguageCode != "" {
		tx = tx.Where("language_code = ?", query.LanguageCode)
	}
	if query.Slug != "" {
		tx = tx.Where("slug = ?", query.Slug)
	}
	if query.Search != "" {
		tx = tx.Where("title ILIKE ?", "%"+query.Search+"%")
	}

	var total int64
	tx.Count(&total)

	if query.OrderBy != "" {
		tx = tx.Order(query.OrderBy)
	} else {
		tx = tx.Order("created_at DESC")
	}
	if query.Limit > 0 {
		tx = tx.Limit(query.Limit)
	}
	if query.Offset > 0 {
		tx = tx.Offset(query.Offset)
	}

	var rows []models.ContentNode
	if err := tx.Find(&rows).Error; err != nil {
		return nil, NewInternal(err.Error())
	}

	nodes := make([]*Node, len(rows))
	for i := range rows {
		nodes[i] = nodeFromModel(&rows[i])

	routeRegistry   = make(map[string]bool) // "METHOD /path" -> true
)

func (c *coreImpl) RegisterRoute(ctx context.Context, method, path string, meta RouteMeta) error {
	if method == "" || path == "" {
		return NewValidation("method and path are required")
	}

	key := fmt.Sprintf("%s %s", method, path)
	routeRegistryMu.Lock()
	defer routeRegistryMu.Unlock()

	if routeRegistry[key] {
		return NewValidation(fmt.Sprintf("route %s already registered", key))
	}

	caller := CallerFromContext(ctx)
	log.Printf("[coreapi] route registered: %s %s (by %s)", method, path, caller.Slug)
	routeRegistry[key] = true

	// Note: Fiber doesn't support dynamic route removal. Routes registered here
	// are tracked for informational purposes. The actual route mounting happens
	// through the script engine's MountRoutes mechanism.
	return nil
}

func (c *coreImpl) RemoveRoute(ctx context.Context, method, path string) error {
	key := fmt.Sprintf("%s %s", method, path)
	routeRegistryMu.Lock()
	defer routeRegistryMu.Unlock()

	if !routeRegistry[key] {
		return NewNotFound("route", key)
	}
	delete(routeRegistry, key)

	log.Printf("[coreapi] route removed: %s", key)
	return nil
}

import (

var filtersMu sync.RWMutex

func (c *coreImpl) RegisterFilter(ctx context.Context, name string, priority int, handler FilterHandler) (UnsubscribeFunc, error) {
	if name == "" {
		return nil, NewValidation("filter name is required")
	}

	entry := filterEntry{priority: priority, handler: handler}

	filtersMu.Lock()
	c.filters[name] = append(c.filters[name], entry)
	sort.Slice(c.filters[name], func(i, j int) bool {
		return c.filters[name][i].priority < c.filters[name][j].priority
	})
	filtersMu.Unlock()

	unsub := func() {
		filtersMu.Lock()
		defer filtersMu.Unlock()
		entries := c.filters[name]
		for i, e := range entries {
			if &e.handler == &handler {
				c.filters[name] = append(entries[:i], entries[i+1:]...)
				break
			}
		}
	}

	return unsub, nil
}

func (c *coreImpl) ApplyFilters(ctx context.Context, name string, value any) (any, error) {
	filtersMu.RLock()
	entries := make([]filterEntry, len(c.filters[name]))
	copy(entries, c.filters[name])
	filtersMu.RUnlock()

	for _, e := range entries {
		value = e.handler(value)
	}
	return value, nil
}

package coreapi

import (
	"context"
	"fmt"
	"io"
	"net/http"
	"strings"
	"time"
)

func (c *coreImpl) Fetch(ctx context.Context, req FetchRequest) (*FetchResponse, error) {
	if req.URL == "" {
		return nil, NewValidation("url is required")
	}
	if req.Method == "" {
		req.Method = "GET"
	}

	timeout := 30 * time.Second
	if req.Timeout > 0 && req.Timeout <= 120 {
		timeout = time.Duration(req.Timeout) * time.Second
	}

	var bodyReader io.Reader
	if req.Body != "" {
		bodyReader = strings.NewReader(req.Body)
	}

	httpReq, err := http.NewRequestWithContext(ctx, req.Method, req.URL, bodyReader)
	if err != nil {
		return nil, NewValidation(fmt.Sprintf("invalid request: %s", err))
	}

	for k, v := range req.Headers {
		httpReq.Header.Set(k, v)
	}

	client := &http.Client{Timeout: timeout}
	resp, err := client.Do(httpReq)
	if err != nil {
		return nil, NewInternal(fmt.Sprintf("fetch failed: %s", err))
	}
	defer resp.Body.Close()

	// Cap response body at 10MB
	body, err := io.ReadAll(io.LimitReader(resp.Body, 10*1024*1024))
	if err != nil {
		return nil, NewInternal(fmt.Sprintf("reading response: %s", err))
	}

	headers := make(map[string]string, len(resp.Header))
	for k := range resp.Header {
		headers[k] = resp.Header.Get(k)
	}

	return &FetchResponse{
		StatusCode: resp.StatusCode,
		Headers:    headers,
		Body:       string(body),
	}, nil
}

		log.Printf("%s INFO: %s%s", prefix, message, fieldStr)

func nodesModule(ctx context.Context, api CoreAPI) map[string]tengo.Object {
	return map[string]tengo.Object{
		"get": &tengo.UserFunction{
			Name: "get",
			Value: func(args ...tengo.Object) (tengo.Object, error) {
				if len(args) != 1 {
					return nil, tengo.ErrWrongNumArguments
				}
				id, ok := tengo.ToInt(args[0])
				if !ok {
					return nil, tengo.ErrInvalidArgumentType{Name: "id", Expected: "int", Found: args[0].TypeName()}
				}
				node, err := api.GetNode(ctx, uint(id))
				if err != nil {
					return wrapError(err), nil

			},
		},

			Name: "get_by_slug",
			Value: func(args ...tengo.Object) (tengo.Object, error) {
				if len(args) != 1 {
					return nil, tengo.ErrWrongNumArguments
				}
				slug, ok := tengo.ToString(args[0])

func httpModule(ctx context.Context, api CoreAPI) map[string]tengo.Object {
	makeRequest := func(method string) *tengo.UserFunction {
		return &tengo.UserFunction{
			Name: strings.ToLower(method),
			Value: func(args ...tengo.Object) (tengo.Object, error) {
				if len(args) < 1 {
					return nil, tengo.ErrWrongNumArguments
				}
				url, ok := tengo.ToString(args[0])
				if !ok {
					return nil, tengo.ErrInvalidArgumentType{Name: "url", Expected: "string", Found: args[0].TypeName()}
				}
				req := FetchRequest{Method: method, URL: url}
				if len(args) > 1 {
					if m, ok := args[1].(*tengo.Map); ok {
						opts := tengoMapToGo(m)
						if h, ok := opts["headers"].(map[string]any); ok {
							req.Headers = make(map[string]string)
							for k, v := range h {
								req.Headers[k] = fmt.Sprintf("%v", v)
							}
						}
						if b, ok := opts["body"].(string); ok {
							req.Body = b
						}
					}
				}
				resp, err := api.Fetch(ctx, req)
				if err != nil {
					return wrapError(err), nil
				}
				return goToTengoObj(map[string]any{
					"status":  resp.StatusCode,
					"headers": resp.Headers,
					"body":    resp.Body,
				}), nil
			},
		}
	}

	// Also provide a generic "fetch" that takes method as first arg
	return map[string]tengo.Object{
		"get":    makeRequest("GET"),
		"post":   makeRequest("POST"),
		"put":    makeRequest("PUT"),
		"delete": makeRequest("DELETE"),
		"fetch": &tengo.UserFunction{
			Name: "fetch",
			Value: func(args ...tengo.Object) (tengo.Object, error) {
				if len(args) < 2 {
					return nil, tengo.ErrWrongNumArguments
				}
				method, _ := tengo.ToString(args[0])
				url, _ := tengo.ToString(args[1])
				req := FetchRequest{Method: strings.ToUpper(method), URL: url}
				if len(args) > 2 {
					if m, ok := args[2].(*tengo.Map); ok {
						opts := tengoMapToGo(m)
						if h, ok := opts["headers"].(map[string]any); ok {
							req.Headers = make(map[string]string)
							for k, v := range h {
								req.Headers[k] = fmt.Sprintf("%v", v)
							}
						}
						if b, ok := opts["body"].(string); ok {
							req.Body = b
						}
					}
				}
				resp, err := api.Fetch(ctx, req)
				if err != nil {
					return wrapError(err), nil
				}
				return goToTengoObj(map[string]any{
					"status":  resp.StatusCode,
					"headers": resp.Headers,
					"body":    resp.Body,
				}), nil
			},
		},
	}
}

// --- Tengo value conversion helpers ---

func wrapError(err error) tengo.Object {
	return &tengo.Error{Value: &tengo.String{Value: err.Error()}}
}

func tengoMapToGo(m *tengo.Map) map[string]any {
	result := make(map[string]any, len(m.Value))
	for k, v := range m.Value {
		result[k] = tengoObjToGo(v)
	}
	return result
}

func tengoObjToGo(obj tengo.Object) any {
	switch o := obj.(type) {
	case *tengo.String:
		return o.Value
	case *tengo.Int:
		return o.Value
	case *tengo.Float:
		return o.Value
	case *tengo.Bool:
		if o == tengo.TrueValue {
			return true
		}
		return false
	case *tengo.Map:
		return tengoMapToGo(o)
	case *tengo.Array:
		result := make([]any, len(o.Value))
		for i, v := range o.Value {
			result[i] = tengoObjToGo(v)
		}
		return result
	case *tengo.Undefined:
		return nil
	default:
		return obj.String()
	}
}

func goToTengoObj(v any) tengo.Object {
	if v == nil {
		return tengo.UndefinedValue
	}
	// Marshal to JSON and back to get a clean interface{} tree,
	// then convert to Tengo objects.
	b, err := json.Marshal(v)
	if err != nil {
		return tengo.UndefinedValue
	}
	var raw any
	if err := json.Unmarshal(b, &raw); err != nil {
		return tengo.UndefinedValue
	}
	return rawToTengo(raw)
}

func rawToTengo(v any) tengo.Object {
	switch val := v.(type) {
	case nil:
		return tengo.UndefinedValue
	case bool:
		if val {
			return tengo.TrueValue
		}
		return tengo.FalseValue
	case float64:
		if val == float64(int64(val)) {
			return &tengo.Int{Value: int64(val)}
		}
		return &tengo.Float{Value: val}
	case string:
		return &tengo.String{Value: val}
	case []any:
		arr := make([]tengo.Object, len(val))
		for i, item := range val {
			arr[i] = rawToTengo(item)
		}
		return &tengo.Array{Value: arr}
	case map[string]any:
		m := make(map[string]tengo.Object, len(val))
		for k, item := range val {
			m[k] = rawToTengo(item)
		}
		return &tengo.Map{Value: m}
	default:
		return &tengo.String{Value: fmt.Sprintf("%v", val)}
	}
}

func applyNodeQueryFromMap(m *tengo.Map, query *NodeQuery) {

		query.Status = v
	}
	if v, ok := gm["language_code"].(string); ok {
		query.LanguageCode = v
	}
	if v, ok := gm["slug"].(string); ok {
		query.Slug = v
	}
	if v, ok := gm["search"].(string); ok {
		query.Search = v
	}

		query.ParentID = &pid
	}

	if err != nil {
		return nil, grpcError(err)
	}
